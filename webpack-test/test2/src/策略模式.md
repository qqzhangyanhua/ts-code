function updateForm2(form, totalRate) {
    if (form === "wb") {
        return totalRate * 0.8;
    } else if (form === "wx") {
        return totalRate * 0.85;

    } else if (form === "h5") {
        return totalRate * 0.88;

    } else if (form === "qq") {
        return totalRate * 0.9;
    }
}







const forms = {
    wb(totalRate) {
        return totalRate * 0.8;
    },
    wx(totalRate) {
        return totalRate * 0.85;
    },
    qq(totalRate) {
        return totalRate * 0.9;
    }
}

function updateForm3(form, totalRate) {
    return forms[form](totalRate)
}
